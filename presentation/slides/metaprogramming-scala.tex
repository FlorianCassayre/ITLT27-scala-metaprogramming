\begin{frame}{Metaprogramming in Scala}

  \begin{itemize}
    \item \textbf{Runtime reflection}: access the high level structure
    \item \textbf{Inlining}: a modifier to guarantee inline at point of use
    \item \textbf{Macros}: converts program code to data, and vice versa
    \item \textbf{Multi-stage programming}: produce code at runtime
    \item \textbf{Compile-time operations}: convert types to values
  \end{itemize}

\end{frame}
