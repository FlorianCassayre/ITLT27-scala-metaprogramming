\begin{frame}{Metaprogramming}

  \begin{block}{}
    A technique in which programs have the ability to
    \textit{analyze}, \textit{generate}, or \textit{transform} other programs---including themselves.
  \end{block}

  \begin{block}{}
    Goals:
  \end{block}
  \begin{itemize}
    \item Increase abstraction
    \item Reduce boilerplate
    \item Improve runtime performance
  \end{itemize}

  \begin{block}{}
    Common challenges:
  \end{block}
  \begin{itemize}
    \item Tooling complexity
    \item Safety
  \end{itemize}

\end{frame}
