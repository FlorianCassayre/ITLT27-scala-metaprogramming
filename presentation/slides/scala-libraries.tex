\begin{frame}{Scala libraries}

  \begin{block}{}
    A non-exhaustive list of libraries exploiting these language features:
  \end{block}

  \begin{itemize}
      \item \href{https://github.com/sbt/sbt}{\textbf{sbt}}: the standard Scala compiler
      \item \href{https://github.com/scalatest/scalatest}{\textbf{scalatest}/\textbf{scalactic}}: debug-friendly assertions
      \item \href{https://github.com/fthomas/refined}{\textbf{refined}}: data constraints represented as types
      \item \href{https://github.com/typelevel/shapeless-3}{\textbf{shapeless}}: a framework that implements \href{https://en.wikipedia.org/wiki/Generic_programming}{\textit{generic programming}}
      \item \href{https://github.com/zio/zio-quill}{\textbf{quill}}: DSL for type-safe SQL queries
      \item \href{https://github.com/softwaremill/magnolia}{\textbf{magnolia}}: automatic \href{https://docs.scala-lang.org/scala3/reference/contextual/derivation.html}{\textit{typeclass derivation}} for ADTs
  \end{itemize}

\end{frame}
